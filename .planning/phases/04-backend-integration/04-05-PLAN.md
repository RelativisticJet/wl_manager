   ```python
   def _action_process_approval(self, request, payload, user, roles):
       """Process approval decision (approve or reject request)."""
       request_id = payload.get("request_id")
       decision = payload.get("decision")  # "approve" or "reject"
       reason = payload.get("reason", "")
       
       if not request_id or decision not in ["approve", "reject"]:
           return self._resp(400, {"error": "Invalid request_id or decision"})
       
       # Delegate to wl_approval — it handles both queue management and replay
       success, message = wl_approval.process_approval(
           session_key=self.session_key,
           analyst=user,
           request_id=request_id,
           decision=decision,
           reason=reason
       )
       
       if success:
           return self._resp(200, {"success": True, "message": message})
       return self._resp(400, {"error": message})
   ```
