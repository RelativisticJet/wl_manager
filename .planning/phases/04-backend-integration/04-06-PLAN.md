    """

    Args:
        csv_file: Full path to CSV

        comment: User's comment for this change
        expected_mtime: For optimistic locking (caller checks; not pipeline's concern)

            "success": bool,
            "message": str (human summary),
            "error": str (if success=false),
            "data": {
                "removed_row_count": int,
                "added_row_count": int,
                "edited_row_count": int,

            }
        }

```python
def _action_save_csv(self, request, payload, user, roles):
    # Parse payload: csv_file, parsed_rows, comment, expected_mtime
    csv_file = payload.get("csv_file")
    parsed_rows = payload.get("parsed_rows", [])
    comment = payload.get("comment", "")
    expected_mtime = payload.get("expected_mtime")
    
    # Optimistic locking check (HTTP-level concern, stays in handler)
    file_stat = os.stat(csv_file)
    if expected_mtime and abs(file_stat.st_mtime - expected_mtime) > 0.01:
        return self._resp(400, "CSV file modified since load; refresh and retry")
    
    # Call pipeline
    result = save_csv_pipeline(csv_file, parsed_rows, comment, expected_mtime,
                               user, self.session_key)
    
    if not result.get("success"):
        return self._resp(400, result.get("error", "Save failed"))
    
    return self._resp(200, {"success": True, "data": result.get("data")})
```
