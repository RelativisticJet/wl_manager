State.on(event, callback)

**Implementation:**

7. `batch()` validates all keys first, then updates all, then fires all events in order
8. `reset()` iterates `_registry`, sets each key to default via `_state[key] = default`, then fires `'state:reset'`

**Implementation details:**
1. Extract URL base from `Splunk.util.make_url('/custom/wl_manager')` pattern
2. Build request URL: `baseUrl + '?action=' + action + '&' + paramString`
3. Use common $.ajax options:

   - `headers: {output_mode: 'json'}`
   - For POST, send `JSON.stringify(payload)` as data
4. Error handler wraps response: `{status: xhr.status, message: extractErrorMessage(xhr.responseJSON), action: action}`
5. Fire `$(document).trigger('wl:restError', [errorObj])` on default error handler

**Concrete implementation pattern:**
```javascript
define(['modules/wl_constants'], function(Constants) {
  "use strict";
  
  var REST = {
    _baseUrl: Splunk.util.make_url('/custom/wl_manager'),
    _errorHandler: null,
    
    setErrorHandler: function(callback) {
      this._errorHandler = callback;
    },
    
    restGet: function(action, params) {
      var self = this;
      var url = this._buildUrl(action, params);
      return $.ajax({
        type: 'GET',
        url: url,
        contentType: 'application/json',
        headers: {output_mode: 'json'},
        traditional: true
      }).fail(function(xhr) {
        if (self._errorHandler) {
          var errorMsg = extractErrorMessage(xhr.responseJSON);
          self._errorHandler({status: xhr.status, message: errorMsg, action: action});
          $(document).trigger('wl:restError', [{status: xhr.status, message: errorMsg, action: action}]);
        }
      });
    },
    
    restPost: function(action, payload) {
      var self = this;
      var url = this._buildUrl(action, {});
      return $.ajax({
        type: 'POST',
        url: url,
        data: JSON.stringify({action: action, ...payload}),
        contentType: 'application/json',
        headers: {output_mode: 'json'}
      }).fail(function(xhr) {
        if (self._errorHandler) {
          var errorMsg = extractErrorMessage(xhr.responseJSON);
          self._errorHandler({status: xhr.status, message: errorMsg, action: action});
          $(document).trigger('wl:restError', [{status: xhr.status, message: errorMsg, action: action}]);
        }
      });
    },
    
    _buildUrl: function(action, params) {
      var url = this._baseUrl + '?action=' + encodeURIComponent(action);
      for (var key in params) {
        if (params.hasOwnProperty(key)) {
          url += '&' + encodeURIComponent(key) + '=' + encodeURIComponent(params[key]);
        }
      }
      return url;
    },
    
    init: function() {
      // Called by entry point to initialize; stub if no wl_ui available yet
    }
  };
  
  return REST;
});
```

  "use strict";
  
  var UI = {
    _messageContainer: null,
    _currentTheme: 'light',
    
    init: function() {
      // Create message container
      this._messageContainer = $('<div>')
        .attr('id', 'wl-messages')
        .css({
          position: 'fixed',
          top: '20px',
          right: '20px',
          zIndex: 10000,
          width: '400px'
        })
        .appendTo('body');
      
      // Detect and apply theme
      var savedTheme = localStorage.getItem('wl_theme_preference');
      if (savedTheme) {
        this._currentTheme = savedTheme;
      } else if (window.matchMedia && window.matchMedia('(prefers-color-scheme: dark)').matches) {
        this._currentTheme = 'dark';
      }
      this._applyTheme();
      
      // Listen to showMsg events
      var self = this;
      $(document).on('wl:showMsg', function(event, type, msg) {
        self.showMsg(type, msg);
      });
    },
    
    showMsg: function(type, message) {
      var cssClass = 'wl-msg wl-msg-' + type;
      var $msg = $('<div>')
        .addClass(cssClass)
        .text(message)
        .appendTo(this._messageContainer);
      
      // Auto-hide for non-error
      if (type !== 'error') {
        setTimeout(function() {
          $msg.fadeOut(function() { $msg.remove(); });
        }, 3000);
      }
    },
    
    showFatalError: function(message) {
      // Create blocking overlay
      var $overlay = $('<div>')
        .addClass('wl-fatal-error-overlay')
        .html('<div class="wl-fatal-error-box"><h2>Fatal Error</h2><p>' + message + '</p><button>Dismiss</button></div>')
        .appendTo('body');
      
      $overlay.find('button').on('click', function() {
        $overlay.remove();
      });
    },
    
    toggleTheme: function() {
      this._currentTheme = this._currentTheme === 'light' ? 'dark' : 'light';
      localStorage.setItem('wl_theme_preference', this._currentTheme);
      this._applyTheme();
    },
    
    _applyTheme: function() {
      if (this._currentTheme === 'dark') {
        $(document.documentElement).addClass('wl-dark');
      } else {
        $(document.documentElement).removeClass('wl-dark');
      }
    }
  };
  
  return UI;
});
