@pytest.mark.slow
@pytest.mark.docker
def test_concurrent_csv_saves_no_corruption(docker_rest_client):
    """Test: 5 threads saving different CSVs simultaneously — no corruption, version manifest consistent."""
    csv_files = [f"test_concurrent_{i}_{int(time.time())}.csv" for i in range(3)]
    
    def save_csv(csv_file):
        docker_rest_client.post_action('save_csv', {
            "csv_file": csv_file,
            "csv_data": f"col1,col2\nval_{random.random()},test\n",
            "comment": f"Concurrent save to {csv_file}"
        })
    
    with ThreadPoolExecutor(max_workers=5) as executor:
        futures = [executor.submit(save_csv, csv_files[i % 3]) for i in range(5)]
        results = [f.result() for f in as_completed(futures)]
    
    # Verify: version manifest intact
    for csv_file in csv_files:
        versions = docker_rest_client.get_action('get_versions', {"csv_file": csv_file})
        assert 'versions' in versions
        # Verify no corrupted manifest JSON
        assert all('timestamp' in v for v in versions['versions'])
```

@pytest.mark.slow
@pytest.mark.docker
def test_approval_race_only_one_succeeds(docker_rest_client, admin_context_1, admin_context_2):
    """Test: 2 admins approve same request simultaneously — only 1 succeeds."""
    # Analyst submits approval request
    analyst_resp = analyst_context.post_action('submit_approval', {...})
    request_id = analyst_resp['data']['request_id']
    
    def approve(admin_ctx):
        try:
            return admin_ctx.post_action('process_approval', {
                "request_id": request_id,
                "action": "approve"
            })
        except Exception as e:
            return {"status": "error", "error": str(e)}
    
    with ThreadPoolExecutor(max_workers=2) as executor:
        results = [executor.submit(approve, ctx).result() for ctx in [admin_context_1, admin_context_2]]
    
    # Verify: exactly 1 succeeded
    success_count = sum(1 for r in results if r.get('status') == 200)
    assert success_count == 1, f"Expected 1 approval, got {success_count}"
```
