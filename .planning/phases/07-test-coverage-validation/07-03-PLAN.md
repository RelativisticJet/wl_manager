import pytest
from hypothesis import given, strategies as st

def test_xss_payloads_sanitized(xss_payloads):
    """Test: all OWASP XSS payloads are sanitized."""
    from bin.wl_validation import sanitize_text
    
    for payload_case in xss_payloads:

        assert '<script>' not in sanitized.lower()
        assert 'onerror=' not in sanitized.lower()
        assert 'onclick=' not in sanitized.lower()
        assert 'javascript:' not in sanitized.lower()
        # Verify: result is not empty (some payloads should still have content)
        if payload.strip():
            assert len(sanitized) > 0

@given(st.text().filter(lambda x: '<' in x))  # Generate strings with < (potential XSS)
def test_xss_fuzzing(payload):
    """Test: Hypothesis-generated XSS-like strings are sanitized."""
    from bin.wl_validation import sanitize_text
    sanitized = sanitize_text(payload)
    # Verify: no raw HTML tags in output
    assert '<' not in sanitized or sanitized.count('<') == 0  # Safe tags only

def test_safe_realpath_prevents_escape(path_traversal_payloads):
    """Test: safe_realpath resolves within lookups/ directory."""
    from bin.wl_validation import safe_realpath
    lookups_dir = '/opt/splunk/etc/apps/wl_manager/lookups'
    
    for payload_case in path_traversal_payloads:
        payload = payload_case['payload']
        try:
            result = safe_realpath(lookups_dir, payload)
            # Verify: result path stays within lookups_dir
            assert result.startswith(lookups_dir), f"Path escapes lookups: {result}"
        except (ValueError, OSError):
            # Expected for malicious paths
            pass

    """Test: CSV header injection attempts are blocked."""
    from bin.wl_validation import sanitize_text
    
    for payload_case in injection_payloads:
        payload = payload_case['payload']
        # CSV headers should not allow pipes, semicolons, newlines
        if '\\n' in payload or '|' in payload or ';' in payload:
            # Should be sanitized or rejected
            sanitized = sanitize_text(payload)
            assert '\\n' not in sanitized or '<no newlines allowed>'

def test_column_name_validation():
