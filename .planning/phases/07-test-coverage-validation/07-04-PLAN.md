from playwright.sync_api import Page
import time

class SplunkPage:
    """Base page object handling Splunk-specific UI quirks."""
    
    def __init__(self, page: Page, base_url="http://localhost:8000"):
        self.page = page
        self.base_url = base_url
    
    def goto(self, path):
        """Navigate to Splunk page."""
        self.page.goto(f"{self.base_url}{path}")
        self.wait_for_splunk_load()
    
    def wait_for_splunk_load(self):
        """Wait for Splunk panels to load."""
        # Splunk injects panels dynamically; wait for visible content
        self.page.wait_for_selector('div[class*="panel"]', timeout=5000)
        time.sleep(0.5)  # Extra buffer for animations
    
    def get_iframe_page(self, panel_name):
        """Get page context within an iframe (for HTML panels)."""
        # Splunk HTML panels are often in iframes
        iframe_locator = f'iframe[name*="{panel_name}"]'
        return self.page.frame_locator(iframe_locator).first

class WhitelistManagerPage(SplunkPage):
    """Main whitelist manager dashboard."""
    
    def load_csv(self, csv_name):
        """Load a CSV file from detection rule dropdown."""
        # Splunk uses custom dropdown components
        dropdown = self.page.locator('div[class*="dropdown"]').first
        dropdown.click()
        self.page.wait_for_selector('text=' + csv_name)
        self.page.click(f'text={csv_name}')
        self.page.wait_for_load_state("networkidle")
    
    def get_table_rows(self):
        """Retrieve current table rows."""
        rows = self.page.locator('table tbody tr').all()
        return [
            {
                'cells': [cell.text_content() for cell in row.locator('td').all()]
            }
            for row in rows
        ]
    
    def edit_cell(self, row_idx, col_idx, new_value):
        """Edit a cell in the table."""
        cell = self.page.locator(f'table tbody tr:nth-child({row_idx}) td:nth-child({col_idx})').first
        cell.click()
        # Activate edit mode (depends on implementation)
        cell.fill(new_value)
        cell.blur()  # Trigger validation/change detection
    
    def add_row(self):
        """Click 'Add Row' button."""
        # Splunk uses span buttons, not <button> elements
        self.page.click('span:has-text("Add Row")')
        self.page.wait_for_selector('input[placeholder="Enter value"]')
    
    def remove_row(self, row_idx, reason="Test removal"):
        """Remove a row with reason."""
        row = self.page.locator(f'table tbody tr:nth-child({row_idx})').first
        checkbox = row.locator('input[type="checkbox"]')
        checkbox.check()
        
        self.page.click('span:has-text("Remove")')
        # Modal appears
        self.page.fill('textarea[name="removal_reason"]', reason)
        self.page.click('span:has-text("Confirm")')
    
    def save_changes(self, comment="E2E test save"):
        """Save changes and submit."""
        self.page.fill('textarea[name="comment"]', comment)
        self.page.click('span:has-text("Save")')
        self.page.wait_for_selector('text="Saved successfully"', timeout=10000)
    
    def get_audit_events(self):
        """Navigate to audit tab and retrieve events."""
        self.page.click('a:has-text("Audit")')
        self.page.wait_for_selector('table', timeout=5000)
        # Return audit event rows (simplified)
        return self.page.locator('table tbody tr').count()

class ControlPanelPage(SplunkPage):
    """Admin control panel."""
    
    def get_approval_queue(self):
        """Retrieve approval queue items."""
        self.page.click('span:has-text("Approval Queue")')
        self.page.wait_for_selector('table')
        items = self.page.locator('table tbody tr').count()
        return items
    
    def approve_request(self, request_idx, comment="Approved by E2E test"):
        """Approve a pending request."""
        row = self.page.locator(f'table tbody tr:nth-child({request_idx})').first
        row.locator('span:has-text("Approve")').click()
        
        self.page.fill('textarea[name="approval_comment"]', comment)
        self.page.click('span:has-text("Submit")')
        self.page.wait_for_selector('text="Approved"', timeout=5000)
    
    def reject_request(self, request_idx, reason="Rejected by E2E test"):
        """Reject a pending request."""
        row = self.page.locator(f'table tbody tr:nth-child({request_idx})').first
        row.locator('span:has-text("Reject")').click()
        
        self.page.fill('textarea[name="rejection_reason"]', reason)
        self.page.click('span:has-text("Submit")')
        self.page.wait_for_selector('text="Rejected"', timeout=5000)

class AuditPage(SplunkPage):
    """Audit dashboard."""
    
    def filter_by_action(self, action):
        """Filter audit by action type."""
        dropdown = self.page.locator('div[class*="action-filter"]').first
        dropdown.click()
        self.page.click(f'text={action}')
        self.page.wait_for_load_state("networkidle")
    
    def get_event_count(self):
        """Get count of audit events displayed."""
        return self.page.locator('table tbody tr').count()
